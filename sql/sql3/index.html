1. Minkälaisella kyselyllä saat listattua kaikki tiettyyn henkilöön liittyvät tilaukset?
SELECT etunimi, sukunimi, tuotetaulu.tuotenimi FROM asiakastaulu
JOIN tilaustaulu
ON asiakastaulu.id = tilaustaulu.asiakas_id
JOIN tilausrivitaulu
ON tilausrivitaulu.tilaus_id = tilaustaulu.id
JOIN tuotetaulu
ON tuotetaulu.id = tilausrivitaulu.tuote_id
WHERE asiakastaulu.id = 27;
2. Minkälaisella kyselyllä saat haettua kaikki myydyt tietyn nimiset tuotteet?
SELECT tilaus_id,tilausmaara, tuotetaulu.tuotenimi, tuotetaulu.hinta
FROM tilausrivitaulu
JOIN tuotetaulu
ON tuotetaulu.id = tilausrivitaulu.tuote_id;
